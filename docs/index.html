Get Started

Add to your project

No package manager. No install. Just copy the files you need.

JavaScript 
// Copy src/prolog-engine.js to your project
import { PrologEngine } from './prolog-engine.js';
const { atom, variable, compound, num } = PrologEngine;

const engine = new PrologEngine();

// Add facts
engine.addClause(compound("temp", [num(85)]));

// Add rules
engine.addClause(
  compound("hot", []),
  [compound("temp", [variable("T")]),
   compound(">", [variable("T"), num(80)])]
);

// Query
engine.queryFirst(compound("hot", []));
// => { type: "atom", name: "hot" }
Python 
# Copy src/prolog.py to your project
from prolog import Engine, atom, var, compound, num

engine = Engine()

# Add facts
engine.add_clause(compound("temp", [num(85)]))

# Add rules
engine.add_clause(
    compound("hot", []),
    [compound("temp", [var("T")]),
     compound(">", [var("T"), num(80)])]
)

# Query
engine.query_first(compound("hot", []))
# => ("compound", "hot", ())
Browser (single HTML file) 
<!-- Inline the engine, import SolidJS -->
<script type="module">
import { createSignal, createMemo }
  from "https://cdn.jsdelivr.net/npm/
    solid-js@1.9.5/+esm";

const engine = new PrologEngine();
const [gen, setGen] = createSignal(0);

// Reactive query: auto-updates
const status = createMemo(() => {
  gen();
  return engine.queryFirst(
    compound("status", [variable("S")])
  );
});

// Change a fact, bump generation
engine.retractFirst(...);
engine.addClause(...);
setGen(g => g + 1);  // status() recomputes
</script>
With Reactive Layer 
# Copy src/reactive.py + src/reactive_prolog.py
from prolog import Engine, atom, var, compound, num
from reactive_prolog import ReactiveEngine
from reactive import effect

engine = Engine()
# ... add clauses ...

rp = ReactiveEngine(engine)

# Reactive query: auto-updates on bump()
status = rp.query_first(
    lambda: compound("status", [var("S")])
)

# Side-effect: runs whenever status changes
def on_change():
    print("Status:", status())
effect(on_change)

# Change facts, bump to recompute
engine.retract_first(...)
engine.add_clause(...)
rp.bump()  # status() recomputes, effect fires
